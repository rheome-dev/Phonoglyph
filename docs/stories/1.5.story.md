### Glassmorphism Component Library
```typescript
// Reusable glass components
export function GlassCard({ 
  children, 
  variant = 'default',
  className,
  ...props 
}: GlassCardProps) {
  const variants = {
    default: 'glass',
    strong: 'glass-strong',
    modal: 'modal-content'
  };
  
  return (
    <div 
      className={cn(variants[variant], className)}
      {...props}
    >
      {children}
    </div>
  );
}

export function GlassModal({ 
  isOpen, 
  onClose, 
  children 
}: GlassModalProps) {
  return (
    <AnimatePresence>
      {isOpen && (
        <motion.div 
          className="modal-overlay"
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
        >
          <motion.div
            className="modal-content"
            initial={{ scale: 0.9, opacity: 0 }}
            animate={{ scale: 1, opacity: 1 }}
            exit={{ scale: 0.9, opacity: 0 }}
          >
            {children}
          </motion.div>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
```
